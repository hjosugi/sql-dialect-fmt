==================
Match_recognize clause with measures, pattern, and define
==================

SELECT * FROM stock_ticks MATCH_RECOGNIZE (
    PARTITION BY symbol
    ORDER BY tick_time
    MEASURES MATCH_NUMBER() AS mno, FIRST(price) AS start_price
    ONE ROW PER MATCH
    AFTER MATCH SKIP TO LAST rise
    PATTERN (dip+ rise+)
    DEFINE dip AS price < LAG(price), rise AS price > LAG(price)
);

==================
Match_recognize written as an immediate call
==================

SELECT price FROM ticks
MATCH_RECOGNIZE(ORDER BY t ALL ROWS PER MATCH PATTERN(a b) DEFINE a AS price > 10, b AS TRUE);
